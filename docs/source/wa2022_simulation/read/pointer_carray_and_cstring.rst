3. 指针运算、C风格数组和C风格字符串
************************************

.. code-block:: cpp
  :linenos:

  #include <iostream>

  void function(char* what, char* ever) {
    int n = 0;
    int m = 0;
    for (; *what != '\0' && *ever != 0; ++what, ++ever) {
      if (*what != *ever) {
        // 永远不用 std::swap(*what, *ever) 的读程序题
        char temp = *what;
        *what     = *ever;
        *ever     = temp;
        ++n;
      } else {
        ++m;
      }
    }

    std::cout << n << ' ' << m << '\n';
    std::cout << what << ' ' << ever << '\n';
  }

  auto main() -> int {
    char what[] = "killermequeen";
    char ever[] = "thirdbmeomb";

    function(what, ever);

    std::cout << what << ' ' << ever << '\n';
  }
