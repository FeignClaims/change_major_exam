5. 递归函数
************************************************************************************************************************

.. code-block:: cpp
  :linenos:

  #include <iostream>

  void function(char* string, int index) {
    if (string[index] == '\0') {
      return;
    }

    char current = string[index];

    int i = 0;
    for (i = index - 1; i >= 0 && *(string + i) > current; --i) {
      string[i + 1] = string[i];
    }
    string[i + 1] = current;

    function(string, index + 1);
  }

  auto main() -> int {
    char string[] = "HelloWorld";
    function(string, 0);
    std::cout << string;
  }
